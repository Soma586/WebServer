<script>

function BCClogin(LoginLocation){
	//alert(LoginLocation);
	LoginLocation = Number(LoginLocation);
	LoginURL = 'http://www.bcc.cuny.edu/';
	switch (LoginLocation){
		case 1:
		// BCC Email
		LoginURL = 'http://www.bcc.cuny.edu/email/';
		break;
		
		case 2:
		// OSSES
		LoginURL = 'http://www.bcc.cuny.edu/OSSES/';
		break;
		
		case 3:
		// CUNYfirst
		LoginURL = 'http://www.bcc.cuny.edu/CUNYfirst/';
		break;
		
		case 4:
		// CUNYportal
		LoginURL = '/CUNYPortal';
		break;
		
		case 5:
		// Blackboard
		//LoginURL = 'http://www.bcc.cuny.edu/blackboard/';
		LoginURL = 'https://bbhosted.cuny.edu/webapps/login/noportal';
		break;
		
		case 6:
		// DegreeWorks
		LoginURL = 'https://degreeworks.cuny.edu/';
		break;

		case 7:
		// BCC One Path
		LoginURL = 'https://bronco.sso.bcc.cuny.edu/';
		break;

		case 8:
		// BCC One Path
		LoginURL = 'https://bcc.starfishsolutions.com/starfish-ops/';
		break;

		default:
		// BCC Website
		LoginURL = 'http://www.bcc.cuny.edu/';
		break;
	}
	//resetLoginSelectElement();
	location.assign(LoginURL);
	
	return;
}
